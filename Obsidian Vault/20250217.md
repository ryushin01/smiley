"use client";  
  
import { NumericFormat, PatternFormat } from "react-number-format";  
  
export default function SampleNumberFormat() {

    <>  
      {/* 통화 */}  
      <NumericFormat  
        // 첫 자리 0 입력 허용  
        // allowLeadingZeros

        suffix={" 원"}

        thousandSeparator=","  
      />

      <PatternFormat  
        // 형식 패턴 설정  
        format="####-##-##"  
        // 표시 금액 값  
        value={20250217}  
        // 표시 타입: text 시 span 렌더링(수정 불가) / input 시 input 렌더링(수정 가능)  
        displayType="text"  
      />  
    </>

}
